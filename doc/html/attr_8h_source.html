315  PYBIND11_NOINLINE void add_base(const std::type_info &base, void *(*caster)(void *)) {
316  auto base_info = detail::get_type_info(base, false);
317  if (!base_info) {
318  std::string tname(base.name());
319  detail::clean_type_id(tname);
320  pybind11_fail("generic_type: type \"" + std::string(name) +
321  "\" referenced unknown base type \"" + tname + "\"");
322  }
323 
324  if (default_holder != base_info->default_holder) {
325  std::string tname(base.name());
326  detail::clean_type_id(tname);
327  pybind11_fail("generic_type: type \"" + std::string(name) + "\" " +
328  (default_holder ? "does not have" : "has") +
329  " a non-default holder type while its base \"" + tname + "\" " +
330  (base_info->default_holder ? "does not" : "does"));
331  }
332 
333  bases.append((PyObject *) base_info->type);
334 
335  if (base_info->type->tp_dictoffset != 0)
336  dynamic_attr = true;
337 
338  if (caster)
339  base_info->implicit_casts.emplace_back(type, caster);
